function launchFireworks(){
for(let i=0;i<120;i++){
let fire=document.createElement("div");
fire.classList.add("firework");

fire.style.left=window.innerWidth/2+"px";
fire.style.top=window.innerHeight/2+"px";

let angle=Math.random()*2*Math.PI;
let radius=Math.random()*250;

fire.style.setProperty('--x',Math.cos(angle)*radius+"px");
fire.style.setProperty('--y',Math.sin(angle)*radius+"px");

document.body.appendChild(fire);
setTimeout(()=>fire.remove(),1000);
}
}
